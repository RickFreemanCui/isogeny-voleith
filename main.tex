\begin{abstract}
	We describe how to use VOLEitH to prove knowledge of an isogeny
	$\phi: E_0 \to E_n$ between two supersingular elliptic curves.
	The prover commits to intermediate $j$-invariants along the isogeny
	path and proves that each consecutive pair satisfies the modular
	polynomial relation $\polyrel_{\ell}(j_{i-1}, j_i) = 0$.
\end{abstract}

% ===================================================================
\section{Introduction}
% ===================================================================

Let $E_0$ and $E_n$ be two supersingular elliptic curves over $\F_{p^2}$.
A prover $\calP$ claims to know an isogeny $\phi: E_0 \to E_n$ of degree
$\ell^k$. To prove this in zero-knowledge using VOLEitH, we decompose
the isogeny into $k$ steps of degree $\ell$ and commit to the intermediate
$j$-invariants. The verifier checks that each consecutive pair satisfies the
$\ell$-th modular polynomial $\polyrel_{\ell}$.

\section{Preliminaries}

\subsection{Modular Polynomials}

\begin{definition}[Classical Modular Polynomial]
	For a prime $\ell$, the $\ell$-th classical modular polynomial
	$\polyrel_{\ell}(X, Y) \in \Z[X, Y]$ is a symmetric polynomial
	of degree $\ell + 1$ in each variable, with the property that for any two
	elliptic curves $E, E'$ over a field of characteristic $\neq \ell$,
	\[
		\polyrel_{\ell}(j(E), j(E')) = 0
		\iff \text{there exists an isogeny } E \to E' \text{ of degree } \ell.
	\]
\end{definition}

For $\ell = 2$, the classical modular polynomial is explicitly:
\begin{equation}\label{eq:modpoly2}
	\begin{aligned}
		\polyrel_2(X, Y) = &\ X^3 + Y^3 + X^2Y^2
		- 1488(X^2Y + XY^2) - 162000(X^2 + Y^2) \\
		&+ 40773375XY + 8748000000(X + Y) - 157464000000000.
	\end{aligned}
\end{equation}
This is a bivariate polynomial of total degree $4$. Evaluating
$\polyrel_2(X, Y)$ requires computing the monomials
$\{X, X^2, X^3, Y, Y^2, Y^3, XY\}$ and taking a linear combination.

For more efficient arithmetization in proof systems, one may use
\emph{canonical modular polynomials}~\cite{canonicalModPoly}.
For $\ell \in \{2, 3, 5, 7, 13\}$, the $\ell$-th canonical modular polynomial
has the structured form
\begin{equation}\label{eq:canonical-modpoly}
	\polyrel_{\ell}^c(X, j) = X^{\ell+1} + \sum_{i=1}^{\ell} c_i X^i
	+ \ell^{s} - X \cdot j,
	\qquad s = \frac{12}{\ell - 1},
\end{equation}
with integer coefficients $c_i \in \Z$ and $c_{\ell+1} = 1$.
Compared to the classical $\polyrel_{\ell}$ (total degree $2\ell$),
$\polyrel_{\ell}^c$ has degree $\ell+1$ in $X$ and degree $1$ in $j$,
which reduces the multiplicative complexity of the verification circuit.

Two curves $E, E'$ are $\ell$-isogenous iff the system
\[
	\polyrel_{\ell}^c(X, j(E)) = 0
	\quad \land \quad
	\polyrel_{\ell}^c(\ell^{s}/X,\, j(E')) = 0
\]
has a common solution $X \in \F_{p^2}$.
Multiplying the second equation by $X^{\ell+1}/\ell^{s}$ yields the equivalent
polynomial $\Theta_{\ell}^c$, giving two explicit degree-$\ell+1$
equations in $X$:
\begin{equation}\label{eq:canonical-two-eq}
	\sum_{i=0}^{\ell+1} c_i X^i - j X = 0
	\quad \land \quad
	\sum_{i=0}^{\ell+1} c'_i X^i - j' X^{\ell} = 0,
\end{equation}
where $c_0 = \ell^{s}$, $c_{\ell+1} = 1$,
$c'_i = c_{\ell+1-i} \cdot \ell^{s(\ell-i)}$, and
$j = j(E)$, $j' = j(E')$.

\subsection{Isogeny Path and Verification}

\begin{definition}[Isogeny Chain]
	An isogeny $\phi: E_0 \to E_k$ of degree $\ell^k$ can be factored as
	\[
		E_0 \xrightarrow{\phi_1} E_1 \xrightarrow{\phi_2} \cdots
		\xrightarrow{\phi_k} E_k
	\]
	where each $\phi_i: E_{i-1} \to E_i$ is an isogeny of degree $\ell$.
	Let $j_i := j(E_i)$. Then $\polyrel_{\ell}(j_{i-1}, j_i) = 0$ for all
	$i \in [1, k]$.
\end{definition}

The prover $\calP$ knows the entire chain, including the intermediate
$j$-invariants $j_1, \dots, j_{k-1}$. The verifier $\calV$ only knows
$j_0 = j(E_0)$ and $j_k = j(E_k)$.

The relation to be proven is:
\begin{equation}\label{eq:isopath-relation}
	R_{\ell^k\text{-}\mathrm{ModPoly}} = \left\{
		((j_0, j_k),\; \vec{j} = (j_1, \dots, j_{k-1})) \;:\;
		\polyrel_{\ell}(j_{i-1}, j_i) = 0,\; \forall i \in [1, k]
	\right\}.
\end{equation}
Equivalently, using canonical polynomials, for each step $i$ the prover
additionally supplies $X_i \in \F_{p^2}$ satisfying
\Cref{eq:canonical-two-eq} with $(j, j') = (j_{i-1}, j_i)$.

\medskip\noindent
\textbf{Verification in the clear.}
To verify an isogeny path without zero-knowledge, $\calV$ evaluates
$\polyrel_{\ell}(j_{i-1}, j_i)$ for $i = 1, \dots, k$ and checks that each
result is $0$. For classical $\polyrel_2$, each evaluation costs a small
constant number of $\F_{p^2}$ multiplications (seven monomials and one
linear combination). For canonical polynomials, each step evaluates two
degree-$\ell+1$ univariate polynomials via Horner's method, costing
$O(\ell)$ multiplications per step.

\medskip\noindent
\textbf{Parameters.}
The protocol operates over $\F_{p^2}$ where $p$ is a large prime of the
form $p = 2^a \cdot 3^b \cdot f \pm 1$, supporting efficient isogeny
computation and FFT-based proof systems~\cite{CLL23}.
The isogeny degree is a small prime $\ell \in \{2, 3, 5, 7, 13\}$.
The path length $k$ must satisfy $\ell^{k/2} \approx 2^{\lambda}$ for
$\lambda$ bits of classical security (claw-finding resistance), i.e.\
$k \approx 2\lambda / \log_2 \ell$. Concrete parameters are listed in
\Cref{tab:isogeny-params}.

For larger $\ell$, the path length shrinks (by a factor of $\log_2 \ell$)
but the per-step verification cost grows as $O(\ell)$. The trade-off is
favorable: moving from $\ell = 2$ to $\ell = 13$ reduces $k$ by a factor
of $\approx 3.7$ while increasing the per-step constraint count by a factor
of $\approx 3.3$~\cite{canonicalModPoly}.

\medskip\noindent
\textbf{Encoding as degree-2 constraints.}
In the VOLEitH setting, each modular polynomial check is expressed as a
small set of degree-2 constraints (multiplication gates) over $\F_{p^2}$.
For classical $\polyrel_2$, each step requires $4$ constraints (squaring
$j_{i-1}$, cubing $j_{i-1}$, computing $j_{i-1} j_i$, and the polynomial
evaluation), giving $4k+2$ constraints for a path of length $k$.
With canonical polynomials, this is reduced to $(\ell+3)k$ constraints
over $\F_{p^2}$; for $\ell = 2$, this becomes $3k$ constraints
(see \Cref{tab:r1cs-comparison} in \Cref{sec:isogeny-proof}).

\subsection{Batch All-but-one Vector Commitment}

A Batch All-but-one Vector Commitment (BAVC) allows a prover to commit
to $\tau$ vectors of seeds, and later open all but one position per vector.
We work over the base field $\F_{p^2}$.
The scheme is parameterized by the number of vectors $\tau$, the vector
length $N = 2^k$, and a leaf commitment function
$\mathsf{LeafCommit}: \{0,1\}^{\secpar} \to \{0,1\}^{2\secpar}$.

The BAVC provides three algorithms~\cite{faestv2}:

\begin{description}[leftmargin=0.9em, labelsep=0.5em]
	\item[$\mathsf{BAVC.Commit}(r) \to (\mathsf{com}, \mathsf{decom}, \{\mathsf{sd}_{i,j}\})$:]
		On input a random root seed $r \in \{0,1\}^\lambda$, builds a GGM tree with $L = \tau \cdot N$ leaves.
		For each leaf position $(i,j) \in [0,\tau) \times [0,N)$, computes
		$(\mathsf{sd}_{i,j}, \mathsf{com}_{i,j}) \gets \mathsf{LeafCommit}(k_\alpha, \cdot)$
		where $k_\alpha$ is the leaf seed.
		Hashes the per-vector commitments $\{\mathsf{com}_{i,j}\}_j$ into
		$h_i$, then outputs the root commitment $\mathsf{com} \gets H(h_0 \| \cdots \| h_{\tau-1})$
		together with full decommitment information $\mathsf{decom}$ and all
		leaf seeds $\{\mathsf{sd}_{i,j}\}$.

	\item[$\mathsf{BAVC.Open}(\mathsf{decom}, I) \to \mathsf{decom}_I$ or $\bot$:]
		Given an index vector $I = (\Delta_0, \dots, \Delta_{\tau-1})$ specifying
		which leaf to hide in each vector, outputs a partial opening $\mathsf{decom}_I$
		that contains the unrevealed leaf commitments and exactly the GGM sibling
		seeds needed to reconstruct all other leaves.
		Aborts with $\bot$ if the required opening exceeds the threshold $T_{\sf open}$.

	\item[$\mathsf{BAVC.Reconstruct}(\mathsf{decom}_I, I) \to (\mathsf{com}, \{\mathsf{sd}_{i,j}\}_{j \neq \Delta_i})$:]
		Reconstructs every leaf except the hidden ones.
		Walks up the GGM tree from the revealed sibling seeds to recover missing
		internal seeds, re-computes leaf commitments, and outputs the recomputed
		commitment $\mathsf{com}$ together with all opened leaf seeds.
\end{description}

BAVC satisfies correctness with aborts:
if $\mathsf{Open}$ returns $\mathsf{decom}_I \neq \bot$, then
$\mathsf{Reconstruct}$ recovers the original $\mathsf{com}$ and the correct
seeds $\mathsf{sd}_{i,j}$ for all $j \neq \Delta_i$.

\subsection{Information-Theoretic Message Authentication Code}

We adopt the generalized Information-Theoretic Message Authentication Code
(IT-MAC) definition of~\cite{EC:ABBS25}, adapted to our setting over~$\F_{p^2}$.

\begin{definition}\label{def:it-mac}
	For $d \in \Z$, a degree-$d$ IT-MAC for $a \in \F_{p^2}$ is
	$\itmac{a}^{(d)} := (p_a(x), k_a, \Delta)$,
	where a prover $\calP$ holds a polynomial ${p}_{a}(x) \in \F_{p^2}[x]$
	with $\deg(p_a) = d$ and leading coefficient $a$;
	while a verifier $\calV$ holds the evaluation
	${k}_a = {p}_{a}(\Delta)$, for a global key
	$\Delta \in \F_{p^2}$ unknown to $\calP$.
	The degree superscript $d$ is omitted when clear from context.
\end{definition}

In bracket notation, we write
$\itmac{a}^{(d)} := (\itmacP{a}^{(d)}, \itmacV{a}^{(d)})$,
where $\itmacP{a}^{(d)}$ denotes the degree-$d$ polynomial $p_a(x)$
held by $\calP$, and $\itmacV{a}^{(d)}$ the evaluation
$k_a = p_a(\Delta)$ held by $\calV$.

Let $\itmac{a}^{(d_a)}$ and $\itmac{b}^{(d_b)}$ be two IT-MACs,
with corresponding polynomials $p_a(x), p_b(x)$ and evaluations $k_a, k_b$,
and assume w.l.o.g., $d_a \geq d_b$.
IT-MAC supports the following local homomorphic operations:
\begin{description}[leftmargin=0.9em, labelsep=0.5em]
	\item[Addition.] To compute $\itmac{a + b}^{(d_a)} = \itmac{a}^{(d_a)} + \itmac{b}^{(d_b)}$,
		$\calP$ computes $p_{a+b}(x) = p_a(x) + p_b(x) \cdot x^{d_a - d_b}$,
		while $\calV$ computes $k_{a+b} = k_a + k_b \cdot \Delta^{d_a - d_b}$.

	\item[Multiplication.] To compute $\itmac{ab}^{(d_a \cdot d_b)} = \itmac{a}^{(d_a)} \cdot \itmac{b}^{(d_b)}$,
		$\calP$ computes $p_{ab}(x) = p_a(x) \cdot p_b(x)$,
		while $\calV$ computes $k_{ab} = k_a \cdot k_b$.

	\item[Constants.] For any $c \in \F_{p^2}$ and $d \in \Z$,
		define $\itmac{c}^{(d)}$ by $p_c(x) = c x^d$ and $k_c = c \Delta^d$.
		Combining this with the above operations, constant addition and
		scalar multiplication preserve degree.
		Subtraction follows as $\itmac{a} - \itmac{b} := \itmac{a} + \itmac{-1}^{(1)} \cdot \itmac{b}$.
		Moreover, to de-randomize a random IT-MAC $\itmac{r}^{(1)}$
		(where $r \sample \F_{p^2}$) into a target value $x \in \F_{p^2}$,
		$\calP$ publishes $\delta := x - r$, and both parties compute
		$\itmac{x}^{(1)} := \itmac{r}^{(1)} + \itmac{\delta}^{(1)}$.
\end{description}

IT-MAC satisfies the following hiding and binding properties:
\begin{description}[leftmargin=0.9em, labelsep=0.5em]
	\item[Hiding.] Random degree-1 IT-MACs are typically generated from VOLE protocols
		(e.g., the one in \Cref{sec:voleith}), which inherently ensure
		that the committed value $a$ is independent of $\calV$'s
		authentication material. Thus, hiding is guaranteed.

	\item[Binding.] To open a committed value $a$, $\calP$ sends its
		(purported) associated polynomial $\tilde{p}_{a'}(x)$ to $\calV$,
		who checks whether $\tilde{p}_{a'}(\Delta) = k_a$.
		Any attempt by $\calP$ to open a different value $a' \neq a$
		while passing this check implies
		$\tilde{p}_{a'}(\Delta) = p_a(\Delta)$.
		Assuming $\Delta \sample \F_{p^2}$, the probability of such a
		collision is at most $\frac{\deg(p_a)}{p^2}$.
\end{description}

We define an additional procedure useful for masking in QuickSilver.
\begin{description}[leftmargin=0.9em, labelsep=0.5em]
	\item[Lifting.] Given a degree-1 IT-MAC 
	$\itmac{r}^{(1)} := (\itmacP{r}^{(1)}, \itmacV{r}^{(1)})$, 
	define $\itmac{0}^{(2)} := \mathsf{Lift}(\itmac{r}^{(1)})$ so that
	$\itmac{0}^{(2)} := (\itmacP{r}^{(1)}, \itmacV{r}^{(1)})$. I.e., 
	the representation remain unchanged, and we merely treat $\itmac{r}^{(1)}$
	as a degree-2 authenticaton of $0 \in \F$ in subsequent computations. 
\end{description}

\section{VOLE-in-the-Head over Large Field}
\label{sec:voleith}
% ===================================================================

The VOLEitH framework has two variants, the 7-round variant specialized
for proving relations over small fields, and a 9-round variant for larger
fields~\cite{baumPubliclyVerifiableZeroKnowledge2023}.
The first variant is (somewhat) widely used in post-quantum digital
signature schemes, e.g., FAEST~v1/2, ReSolveD, and SDitH~v2 all use VOLEitH
over $\FF_{2}$.
In this write-up we focus on the 9-round variant.

\subsection{Parameters and Setup}

The prover $\calP$ holds a witness $\vec{w} \in \F^{\ell_{\sf wit}}$ that satisfies
$f_i(\vec{w}) = 0$ for all $i \in [t]$.
The verifier $\calV$ knows the constraints $\{ f_i \}_{i \in [t]}$ but not the witness.

For $n \in \mathbb{N}^+$, we define $[n] := \{ 0, 1, ..., n-1 \}$.

We define $\ell_{\sf vole} := \lceil \frac{\ell_{\sf wit} + 1}{k_{\calC}} \rceil$.
$\calP$ arranges the witness $\vec{w}$ and a random $\F$ element $r_{\sf qs}$ into a $\ell_{\sf vole} \times k_{\calC}$ matrix over $\F$ as $\mat{W}$.
In practice, we can set $r_{\sf qs}$ to be part of the $\mat{U}$ matrix and save sending one $\F$ element in $\mat{D}$.
The internal QuickSilver proof system works on $\ell_{\sf vole} \cdot k_{\calC}$ random IT-MAC relations.
Define $\hat{\ell}_{\sf vole} := 2 \cdot \ell_{\sf vole} + 1$.
The VOLEitH framework generates $\hat{\ell}_{\sf vole}$ subspace VOLE correlations over $\F$ and we sacrifice one to check consistency.



The protocol is parameterized by:
\begin{itemize}
	\item A security parameter $\secpar$ and a field $\F = {GF}(p^2)$.
	\item A pseudorandom generator $\mathsf{PRG}: \{0,1\}^{\secpar} \to \F^{\hat{\ell}_{\sf vole}}$.
	\item A number $\tau = n_C$ of parallel all-but-one VOLE instances and
		a leaf count $N = 2^k$ per instance.
	\item A systematic $[n_{\calC}, k_{\calC}, d_{\calC}]$-linear error correction code
		$\calC \subseteq \F^{n_C}$, with encoding algorithm
		$\mathsf{Enc}_{\calC}$.
	\item A small subset
		$\calD = \{\alpha_0, \dots, \alpha_{N-1}\} \subset \F$
		of size $N = 2^k$, for converting all-but-one vector commitment into VOLE correlations.
	\item A set of degree-2 constraints $f_0, \dots, f_{t-1}$ over $\F$ to be proven. For each $i \in [1, t]$, the witness $\vec{w}$ should satisfy that $f_i (\vec{w}) = 0$.
\end{itemize}

		& & BAVC opening and SoftSpokenOT checks: 
		\begin{itemize}
			\item $\mathsf{com}^{\star} = \mathsf{com}$
			\item $\sum_{i} (\chi_1)^{i} \cdot \mat{Q}[i, *] = \mathsf{Enc}_{\calC}(\tilde{\vec{u}}) \cdot \mat{\Delta} + \tilde{\vec{v}}$
		\end{itemize} \\
		& & QuickSilver check:
		\begin{itemize}
			\item Parse $\mat{Q}'$ as $\itmacV{\mat{W}}^{(1)}$
			\item $\sum_{i} f_i (\itmacV{\vec{w}}^{(1)}) + \mathsf{Lift}(\itmacV{r_{\sf qs}}^{(1)}) = \itmacV{z}^{(2)}(\Delta')$
		\end{itemize} \\
		& &  \\
		& & Subspace VOLE opening check:
		\begin{itemize}
			\item $\mat{Q}_1 := \mat{Q}[[\ell_{\sf vole}], *] + \mathsf{Enc}_{\calC}(\mat{D}) \cdot \mat{\Delta}$
			\item $\mat{Q}_2 := \mat{Q}[[\ell_{\sf vole}, 2\ell_{\sf vole}), *]$
			\item $\mat{Q}_1 \cdot \Delta' + \mat{Q}_2 = \mathsf{Enc}_{\calC}(\mat{Q}') \cdot \mat{\Delta} + \mathsf{M}[\mat{Q}']$
		\end{itemize} \\
		\bottomrule
	\end{tabular}

	\caption{The 9-round VOLEitH Protocol.}
	\label{fig:9-round-voleith-2}
\end{figure}

\subsection{Round-by-Round Explanation}

\begin{description}[leftmargin=0.9em, labelsep=0.5em]
	\item[Commitment and subspace VOLE.]
		$\calP$ commits to $\tau=n_C$ seed vectors with
		$\mathsf{BAVC.Commit}$.  Each vector is converted, via the
		SoftSpokenOT summation over $\calD$, into one raw coordinate of a
		VOLE transcript.  The prover takes the first $k_C$ raw coordinates
		as $\mat{U}$ and sends
		$\mat{C} = \mathsf{Enc}_{\calC}(\mat{U})-\widetilde{\mat{U}}$
		so that $\calV$ can correct the opened raw subspace VOLE materials,
		into authentication of Reed-Solomon codewords.

	\item[Committing to the witness by subspace VOLE.]
		$\calP$ uses the first $\ell_{\sf vole}$ rows of $\mat{U}$ to commit to the witness
		$\mat{W}$, allowing $\calP$ and $\calV$ to get the subspace VOLE commitment 
		of $\vec{w}$ and the mask $r_{\sf qs}$.

	\item[SoftSpokenOT Checking.]
		$\calV$ needs to check that the correction matrix $\mat{C}$ indeed fixes $\tilde{\mat{U}}$
		into the requried format, i.e., every row is a codeword of $\calC$. So using the challenge
		$\chi_1$, $\calP$ computes the linear combination of the rows of $\mat{U}$ and sends result to 
		$\calV$. $\tilde{\vec{u}}$ is masked by $\mat{U}[2\ell_{\sf vole} + 1, *]$, sending $\tilde{\vec{u}}$
		does not leak information about $\mat{W}$.

	\item[Batched QuickSilver Proof.]
		$\calP$ convinces $\calV$ that $\forall i \in [t]$, $f_{i} (\vec{w}) = 0$.
		This is done by evaluating $f_{i}$ on the IT-MAC commitment of $\vec{w}$,
		and sending the masked polynomial $\itmacP[z]^{(2)}$ to $\calV$, who then 
		checks that it indeed evaluates to $\itmacV[z]^{(2)}$ on the IT-MAC global key.
		But since subspace VOLE cannot support arithmetic within a committed vector,
		an additional step is to simulate regular IT-MACs within subspace VOLE.
		In particular, after $\calP$ sends its QuickSilver messages, $\calV$ samples an 
		IT-MAC key $\Delta'$ and sends it to $\calP$.
		$\calP$ uses the first $\ell_{\sf vole}$ rows of $\mat{U}$, $\mat{W}$, as IT-MAC messages and 
		the next $\ell_{\sf vole}$ rows of $\mat{U}$, $\mat{V}'$, as IT-MAC tags. 
		Then, $\calP$ directly computes $\mat{Q}' = \mat{W} \cdot \Delta' + \mat{V}'$ and 
		sends $\mat{Q}'$ as well as its subspace VOLE authentication material $\mathsf{M}[\mat{Q}']$
		to $\calV$.
		The security of subspace VOLE prevents $\calP$ from cheating in the computation of 
		$\mat{Q}'$, and ensures that $\calV$ can use the received $\mat{Q}'$ as the correct 
		IT-MAC local key.

	\item[BAVC opening and verification.]
		The final challenge is a vector $I := (\delta_0, ..., \delta_{n_{\calC} - 1})$
		of $\tau$ random indices in $[N]$. 
		$\calP$ opens the BAVC commitment so that for $i \in [n_{\calC}]$, 
		$\calV$ learns $\mathsf{sd}_{i, j}$ for $j \in [N], j \neq \delta_{i}$.
		Using the same summation method, $\calV$ can compute the ``raw'' subspace VOLE 
		authentication materials $\widetilde{\mat{Q}}$.
\end{description}

From $\widetilde{\mat{Q}}$, $\calV$ can run the following checks:
\begin{itemize}
	\item \textbf{BAVC Opening Check:} Verify that the reconstructed $\mathsf{com}^{\star}$
	indeed matches the $\mathsf{com}$ that $\calP$ sends. This ensure that BAVC opening is 
	correct. 
	
	\item \textbf{SoftSpokenOT Check: } Verify that the $(\tilde{\vec{u}}, \tilde{\vec{v}})$ pair 
	is need a valid subspace VOLE correlation under the global key $\mat{\Delta}$ (defined by $I$).
	This ensures that the $\mat{C}$ sent by $\calP$ is correct. 

	\item \textbf{QuickSilver Check:} Verify that the $\itmacP{z}^{(2)}$ sent by $\calP$ evaluates 
	to $\itmacV{z}^{(2)}$ at $\Delta'$, where $\calV$ uses $\mat{Q}'$ to define 
	$\itmacV{\mat{W}}^{(1)}$, under $\Delta'$. 
	This ensure that the constraints $f_0, ..., f_{t-1}$ are satisfied 
	by the committed $\vec{w}$.

	\item \textbf{Subspace VOLE Opening Check:} Verify that every row of the sent 
	$(\mat{Q}', \mathsf{M}[\mat{Q}'])$ is indeed a valid subspace VOLE correlation 
	under the gloabl key $\mat{\Delta}$. This ensures that the $\mat{Q}'$ in the previous 
	step is indeed correct.
\end{itemize}

\subsection{Parameters for $\lambda = 128$}

while also reducing the witness size from $256$ to $70$ intermediate
$j$-invariants.

\textbf{Implications for VOLEitH.}
The VOLEitH proof system of \Cref{sec:voleith} scales with the number of
degree-2 constraints $t$ and the witness size $\ell_{\sf wit}$.
With canonical modular polynomials at $\ell = 2$ over $\F_{p^2}$, we have
$t = 3k = 768$ constraints and $\ell_{\sf wit} = 3k+1 = 769$ witness
elements for $\lambda = 128$.
Using $\ell = 13$ reduces the witness to $701$ elements but increases the
per-step constraint count. The optimal choice depends on the concrete
VOLEitH parameters (code $\calC$, VOLE batch size $\tau$, etc.),
which we instantiate in the next subsection.

\section{Security Analysis}
% ===================================================================

\begin{theorem}[Completeness]
	If $\calP$ knows a valid isogeny $\phi: E_0 \to E_n$, then the honest
	verifier accepts.
\end{theorem}

\begin{proof}
	By definition of the modular polynomial, each consecutive pair
	$(j_{i-1}, j_i)$ along the decomposed isogeny path satisfies
	$\polyrel_{\ell}(j_{i-1}, j_i) = 0$. Since VOLEitH commitments are
	computationally binding and the linear combination $s_i$ is opened
	correctly, all checks pass.
\end{proof}

\begin{theorem}[Soundness]
	If $E_0$ and $E_n$ are not connected by an isogeny of degree $\ell^k$,
	then a malicious prover $\calP^*$ cannot make $\calV$ accept except with
	negligible probability.
\end{theorem}

\begin{proof}[Proof sketch]
	If the verifier accepts, then by the binding property of VOLE
	commitments and the soundness of the linear opening, there exist
	values $j_1, \dots, j_{k-1}$ such that
	$\polyrel_{\ell}(j_{i-1}, j_i) = 0$ for all $i$.
	By the property of modular polynomials, this implies the existence
	of $\ell$-isogenies between consecutive curves, and hence a composite
	isogeny $E_0 \to E_n$ of degree $\ell^k$.
\end{proof}

\begin{remark}
	Knowledge soundness follows from the extractability of VOLEitH
	commitments: an extractor can recover $j_1, \dots, j_{k-1}$ by
	rewinding the prover and solving the resulting linear system.
\end{remark}
